cmake_minimum_required(VERSION 3.16)
project(cannnddyy)

set(CMAKE_CXX_STANDARD 17)

find_package(Qt6 COMPONENTS Core Widgets WebEngineWidgets WebChannel REQUIRED)

add_executable(cannnddyy
    main.cpp
    browserwindow.cpp
    browserwindow.h
    settingsbridge.cpp
    settingsbridge.h
)

target_link_libraries(cannnddyy PRIVATE Qt6::Core Qt6::Widgets Qt6::WebEngineWidgets Qt6::WebChannel)

# copy pages folder into build/cannnddyy/pages
file(COPY ${CMAKE_SOURCE_DIR}/pages DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/cannnddyy)
